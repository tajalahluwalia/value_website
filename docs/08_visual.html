import pandas as pd
import numpy as np
import plotly.express as px

grp_country_yr = pd.read_csv('grp_country_yr.csv')
fig = px.line(grp_country_yr[grp_country_yr['country'].isin(select_country)]
        , x="year", y="secular"
        , hover_name= 'country'             
        , color='country',markers=True
        )

fig.add_hrect(
        y0=0, y1=0.75, line_width=0, 
        fillcolor="green", opacity=0.05,
        annotation_text="Secular", annotation_position="top",
        )
fig.add_hrect(
        y0=0, y1=-0.75, line_width=0, 
        fillcolor="red", opacity=0.05,
        annotation_text="Traditional", annotation_position="bottom",
        )
# fig.update_traces(visible='legendonly')
fig.update_yaxes(visible=False)
fig.update_yaxes(showticklabels=False)
fig.show()
