grp_country_age_sex_yr = pd.read_csv('grp_country_age_sex_yr.csv')
fig = px.line(grp_country_age_sex_yr[grp_country_age_sex_yr['country'].isin(select_country)]
        , x="year", y="secular"
        , hover_name= 'country'             
        , color='country',markers=True, title='Please Select Countries'
        , facet_col = 'sex')

fig.add_hrect(
        y0=0, y1=1.5, line_width=0, 
        fillcolor="green", opacity=0.05,
        annotation_text="Secular", annotation_position="top",
        )
fig.add_hrect(
        y0=0, y1=-1.5, line_width=0, 
        fillcolor="red", opacity=0.05,
        annotation_text="Traditional", annotation_position="bottom",
        )
fig.update_traces(visible='legendonly')
fig.show()
