1 Understanding the Factors

1.1 Traditional Values (Secular values emphasizing the opposite [secular = positive pole]):

Code 
import pandas as pd
import numpy as np
import plotly.express as px

# PLOT
df = df.sort_values(by = 'secular')
fig = px.bar(df,y=df.iloc[:,0],x="secular", orientation='h')
fig.add_hrect(
        y0=9, y1=18, line_width=0, 
        fillcolor="green", opacity=0.1,
        annotation_text="Secular", annotation_position="top left",
        )
fig.add_hrect(
        y0=9, y1=-1, line_width=0, 
        fillcolor="red", opacity=0.1,
        annotation_text="Traditional", annotation_position="bottom right",
        )

fig.show()

1.2 Survival Values (Self-expression values emphasizing the opposite [self-expression = positive pole]):

Code 
import pandas as pd
import numpy as np
import plotly.express as px

# PLOT
df = df.sort_values(by = 'self_expression')
fig = px.bar(df.sort_values(by = 'self_expression'),y=df.iloc[:,0],x="self_expression", orientation='h')
fig.add_hrect(
        y0=9, y1=18, line_width=0, 
        fillcolor="green", opacity=0.1,
        annotation_text="Self Expression", annotation_position="top left",
        )
fig.add_hrect(
        y0=9, y1=-1, line_width=0, 
        fillcolor="red", opacity=0.1,
        annotation_text="Survival", annotation_position="bottom right",
        )

fig.show()
